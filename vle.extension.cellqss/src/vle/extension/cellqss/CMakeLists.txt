INCLUDE_DIRECTORIES(${CMAKE_SOURCE_DIR}/src ${VLE_INCLUDE_DIRS}
  ${CELLDEVS_INCLUDE_DIRS} ${Boost_INCLUDE_DIRS} ${CMAKE_BINARY_DIR}/src)

LINK_DIRECTORIES(${VLE_LIBRARY_DIRS} ${CELLDEVS_LIBRARY_DIRS}
  ${Boost_LIBRARY_DIRS})

ADD_LIBRARY(cellqss STATIC CellQSS.cpp CellQSS.hpp)

SET_TARGET_PROPERTIES(cellqss PROPERTIES OUTPUT_NAME
  "cellqss-${MODEL_MAJOR}.${MODEL_MINOR}")

IF("${CMAKE_SYSTEM_PROCESSOR}" STREQUAL "x86_64")
  if (CMAKE_COMPILER_IS_GNUCC AND CMAKE_COMPILER_IS_GNUCXX)
    SET_TARGET_PROPERTIES(cellqss PROPERTIES COMPILE_FLAGS -fPIC)
  endif ()
ENDIF("${CMAKE_SYSTEM_PROCESSOR}" STREQUAL "x86_64")

TARGET_LINK_LIBRARIES(cellqss ${CELLDEVS_LIBRARIES} ${VLE_LIBRARIES}
  ${Boost_LIBRARIES})

INSTALL(TARGETS cellqss ARCHIVE DESTINATION lib)

CONFIGURE_FILE(Version.hpp.in
  ${CMAKE_BINARY_DIR}/src/vle/extension/cellqss/Version.hpp)

INSTALL(FILES
  ${CMAKE_BINARY_DIR}/src/vle/extension/cellqss/Version.hpp
  DESTINATION src/vle/extension/cellqss)

INSTALL(FILES CellQSS.hpp
  DESTINATION src/vle/extension/cellqss)
