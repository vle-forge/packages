INCLUDE_DIRECTORIES(${CMAKE_SOURCE_DIR}/src ${VLE_INCLUDE_DIRS}
  ${Boost_INCLUDE_DIRS} ${CMAKE_BINARY_DIR}/src)

LINK_DIRECTORIES(${VLE_LIBRARY_DIRS} ${Boost_LIBRARY_DIRS})

ADD_LIBRARY(petrinet STATIC PetriNet.cpp PetriNet.hpp)

IF("${CMAKE_SYSTEM_PROCESSOR}" STREQUAL "x86_64")
  if (CMAKE_COMPILER_IS_GNUCC AND CMAKE_COMPILER_IS_GNUCXX)
    SET_TARGET_PROPERTIES(petrinet PROPERTIES COMPILE_FLAGS -fPIC)
  endif ()
ENDIF("${CMAKE_SYSTEM_PROCESSOR}" STREQUAL "x86_64")

SET_TARGET_PROPERTIES(petrinet PROPERTIES OUTPUT_NAME
  "petrinet-${MODEL_MAJOR}.${MODEL_MINOR}")

TARGET_LINK_LIBRARIES(petrinet ${VLE_LIBRARIES})

INSTALL(TARGETS petrinet ARCHIVE DESTINATION lib)

install(FILES PetriNet.hpp
  DESTINATION src/vle/extension/petrinet)

CONFIGURE_FILE(Version.hpp.in
  ${CMAKE_BINARY_DIR}/src/vle/extension/petrinet/Version.hpp)

INSTALL(FILES
  ${CMAKE_BINARY_DIR}/src/vle/extension/petrinet/Version.hpp
  DESTINATION src/vle/extension/petrinet)
